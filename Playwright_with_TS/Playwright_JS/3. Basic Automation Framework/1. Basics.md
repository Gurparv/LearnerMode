## Installation
```Bash
npm init playwright@latest

//To Install browser
npx playwright install
```

## To Run
```Bash
npx playwright test
```

## Writing your first Script
```Typescript title="test1.spec.js"
import {test,expect} from '@playwright/test';

test('My First Test',async({page})=>{
	console.log("Test starting...")
	await page.goto("https://playwright.dev/docs/api/class-pageassertions#page-assertions-to-have-title");
	console.log(await page.title());
	await expect(page).toHaveTitle("PageAssertions | Playwright");
	
});
```

## Writing your own 'Playwright.config' file 
#### 1. Adding Global configuration Options
Documentation -> https://playwright.dev/docs/api/class-testconfig

```Typescript title="playwright.config.ts"
import {defineConfig} from '@playwright/test';

export default defineConfig({
	timeout: 30_000,
	globalTimeout: 60_000,
	reporter: 'html',
	testDir: './tests',
	fullyParallel: false,

});
```

#### 2. Adding 'use' TestOptions in Config file
Documentation -> https://playwright.dev/docs/api/class-testoptions

```Typescript title="playwright.config.ts"
import {defineConfig} from '@playwright/test';

export default defineConfig({
	use: {
		headless: false,
		viewport:{width: 1280, height: 720},
		browserName: 'firefox',
	}
});

```

#### 3. Adding 'Project' settings in Config file
Documentation -> https://playwright.dev/docs/api/class-testproject

```Typescript title="playwright.config.ts"
import {defineConfig, devices} from '@playwright/test';

export default defineConfig({
	use:{headless:false},
	workers: 1,

	projects : [
		{
			name: 'chromium',
			use: devices['Desktop Chrome'],
		},
		{
			name: 'firefox',
			use: devices['Desktop Firefox'],
		},
		{
			name: 'Mobile Chrome',
			use: devices['Pixel 5'],
		},
	],

});


```

## Getting values about `Test and Config` from Tests at runtime.

Documentation for:
Get Current Running Test Info -> https://playwright.dev/docs/api/class-testinfo
Get Global Config info -> https://playwright.dev/docs/api/class-fullconfig
Get 'Project' Config info -> https://playwright.dev/docs/api/class-fullproject
Get 'use' Config info -> https://playwright.dev/docs/api/class-testconfig


```Typescript title="test1.spec.ts"
import {test,expect} from '@playwright/test';

test('My First Test',async({page},testInfo)=>{
	// Get info about current running test.
	console.log(`The title of the test is: ${testInfo.title}`);
	console.log(`The duration of test is: ${testInfo.duration}`);

	//Get info about Global Config
	console.log(`Path to config file: ${testInfo.config.configFile}`);
	console.log(`Config FullyParallel setting: ${testInfo.config.fullyParallel}`);
	
	// Get info about 'project' config
	console.log(`Dependencies of project: ${testInfo.project.dependencies}`);
	console.log(`grep of project: ${testInfo.project.grep}`);

	// Get info about 'use' config
	console.log(`BaseURl in project: ${testInfo.project.use.baseURL}`);
	console.log(`Mobile in project: ${testInfo.project.use.isMobile}`);
	console.log(`Storage state used in project: ${testInfo.project.use.storageState}`);
	console.log(`Viewport in project: ${testInfo.project.use.viewport}`);
	
});
```
