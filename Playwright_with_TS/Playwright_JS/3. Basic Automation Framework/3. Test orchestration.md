## Writing Basic Test
```TypeScript title="test1.spec.ts"
import {test,expect} from '@playwright/test';

test('basic test',async({page})=>{
	await page.goto('https://google.com');	
});
```

## Adding Tags to test

```TypeScript title="test1.spec.ts"
import {test,expect} from '@playwright/test';

test('basic test',{
	tag: ['@smoke','@sanity'],	
},async({page})=>{
	await page.goto('https://google.com');	
});

test('basic test2',{
	tag: '@smoke',
},async({page})=>{
	await page.goto('https://google.com');	
});

```

Running Tags test
```Bash
npx playwright test --grep "@smoke"
```

You can also filter tests by their tags during test execution:

- in the [command line](https://playwright.dev/docs/test-cli#all-options);
- in the config with [testConfig.grep](https://playwright.dev/docs/api/class-testconfig#test-config-grep) and [testProject.grep](https://playwright.dev/docs/api/class-testproject#test-project-grep);

Learn more about [tagging](https://playwright.dev/docs/test-annotations#tag-tests).

## Adding Annotations
```Typescript title="test1.spec.ts"
import {test,expect} from '@playwright/test';

test('Basic test',
     {
	     annotation: {
		     type: 'issue',
		     description: 'Description of the issue',
	     },
     },async({page})=>{
	await page.goto("https://google.com");
});

```

## Grouping Tests

```TypeScript title="test1.spec.ts"
import {test,expect} from '@playwright/test';

test.describe('UI Tests',()=>{
	test('Test1',async({page})=>{
		await page.goto('https://google.com');
	});

	test('Test2',async({page})=>{
		await page.goto('https://saucedemo.com');
	});
});

```

## Running Specific test(s) for debugging

```TypeScript title="test1.spec.ts"
import {test,expect} from '@playwright/test';

test.only('Trying to debug the issue',async({page})=>{
	await page.goto("https://instagram.com");
});

```
## Skipping Tests

```TypeScript title="test1.spec.ts"
import {test,expect} from '@playwright/test';

test.skip('Trying to debug the issue',async({page})=>{
	await page.goto("https://instagram.com");
});

```

## Specify test which is expected to 'fail'

```TypeScript title="test1.spec.ts"
import {test,expect} from '@playwright/test';

test.fail('this test should fail',async({page})=>{
	await page.goto("https://instagram.com");
	await expect(page).toHaveTitle('SnapChat');
});

```
