# 🔹 Playwright Utility (Node.js, Recommended)

### ✅ Install

```bash
npm install pidusage
```

### ✅ `memoryTracker.ts`

```ts
import pidusage from 'pidusage';

export class MemoryTracker {
  private pid: number;
  private baselineMB: number = 0;

  constructor(private browser: any, private page: any) {}

  async init() {
    this.pid = this.browser.process().pid;
    this.baselineMB = await this.getProcessMemoryMB();
  }

  async getProcessMemoryMB(): Promise<number> {
    const stat = await pidusage(this.pid);
    return stat.memory / 1024 / 1024; // bytes → MB
  }

  async getJSHeapMB(): Promise<number> {
    const client = await this.page.context().newCDPSession(this.page);
    const metrics = await client.send('Performance.getMetrics');

    const heap = metrics.metrics.find((m: any) => m.name === 'JSHeapUsedSize');
    return heap ? heap.value / 1024 / 1024 : 0;
  }

  async report(testName: string) {
    const finalMB = await this.getProcessMemoryMB();
    const jsHeapMB = await this.getJSHeapMB();

    const delta = finalMB - this.baselineMB;

    console.log(`\n🧠 ${testName}`);
    console.log(`Process Memory: ${finalMB.toFixed(0)} MB`);
    console.log(`Delta: +${delta.toFixed(0)} MB`);
    console.log(`JS Heap: ${jsHeapMB.toFixed(0)} MB\n`);

    return {
      totalMB: finalMB,
      deltaMB: delta,
      jsHeapMB
    };
  }
}
```

```ts
import pidusage from 'pidusage';
import fs from 'fs';

export class MemoryTracker {
  private pid: number;
  private baselineMB: number = 0;
  private results: any[] = [];

  constructor(private browser: any, private page: any) {}

  async init() {
    this.pid = this.browser.process().pid;
    this.baselineMB = await this.getProcessMemoryMB();
  }

  async getProcessMemoryMB(): Promise<number> {
    const stat = await pidusage(this.pid);
    return stat.memory / 1024 / 1024;
  }

  async getJSHeapMB(): Promise<number> {
    const client = await this.page.context().newCDPSession(this.page);
    const metrics = await client.send('Performance.getMetrics');
    const heap = metrics.metrics.find((m: any) => m.name === 'JSHeapUsedSize');
    return heap ? heap.value / 1024 / 1024 : 0;
  }

  async capture(testName: string) {
    const finalMB = await this.getProcessMemoryMB();
    const jsHeapMB = await this.getJSHeapMB();
    const delta = finalMB - this.baselineMB;

    const record = {
      testName,
      totalMB: Math.round(finalMB),
      deltaMB: Math.round(delta),
      jsHeapMB: Math.round(jsHeapMB),
      timestamp: new Date().toISOString()
    };

    this.results.push(record);

    console.log(`🧠 ${testName}: +${record.deltaMB} MB`);
    return record;
  }

  saveToFile(file = 'memory-results.json') {
    fs.writeFileSync(file, JSON.stringify(this.results, null, 2));
  }
}
```

# 🔥 1. Heap Snapshot Capture (Playwright + CDP)

This lets you answer:

> “What objects increased between Step A → Step B?”

### ✅ `heapProfiler.ts`

```ts
import fs from 'fs';

export class HeapProfiler {
  private client: any;

  constructor(private page: any) {}

  async init() {
    this.client = await this.page.context().newCDPSession(this.page);
  }

  async takeSnapshot(label: string) {
    const chunks: string[] = [];

    this.client.on('HeapProfiler.addHeapSnapshotChunk', (chunk: any) => {
      chunks.push(chunk.chunk);
    });

    await this.client.send('HeapProfiler.enable');
    await this.client.send('HeapProfiler.takeHeapSnapshot', { reportProgress: false });

    const file = `heap-${label}.heapsnapshot`;
    fs.writeFileSync(file, chunks.join(''));

    console.log(`📸 Heap snapshot saved: ${file}`);
  }
}
```

---

### ✅ Usage

```ts
const heap = new HeapProfiler(page);
await heap.init();

await heap.takeSnapshot('before');

// run suspicious step
await page.click('#load-heavy-data');

await heap.takeSnapshot('after');
```

# 🔥 4. Automatic Leak Detection Heuristic

Add a simple but powerful signal:

### ✅ Leak Detector

```ts
let previous = 0;

export async function detectLeak(page, label) {
  const client = await page.context().newCDPSession(page);
  const metrics = await client.send('Performance.getMetrics');

  const heap = metrics.metrics.find(m => m.name === 'JSHeapUsedSize');
  const current = heap.value / 1024 / 1024;

  if (current > previous * 1.5) {
    console.warn(`🚨 Possible leak at ${label}: ${current.toFixed(0)} MB`);
  }

  previous = current;
}
```
