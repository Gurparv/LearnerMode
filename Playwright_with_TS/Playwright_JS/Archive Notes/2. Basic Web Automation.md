### Extracting the text from browser and inserting assertion.
```example.spec.js
const {test} = require(`@playwright/test`);
const {expect} =  require(`@playwright/test`);


test.only ('First Playwright TestCase', async ({browser,page})=> //test.only means just run this test even if this file contains 20 testcases in total
	{
		await page.goto("https://rahulshettyacademy.com/loginpagePractise/");
		await page.locator('#username').type('rahulshetty');
		await page.locator("[type='password']").type("learning");
		await page.locator("#signInBtn").click();
		const errMsg = await page.locator("[style*='block']").textContent(); //getting text from browser
		console.log(errMsg); 
		await expect(page.locator("[style*='block']")).toContainText('Incorrect username'); //adding assertion
	}
);
```
### How to clear data from browser text field
```example.spec.js
const {test} = require(`@playwright/test`);
const {expect} =  require(`@playwright/test`);


test.only ('First Playwright TestCase', async ({browser,page})=>
	{
		await page.goto("https://rahulshettyacademy.com/loginpagePractise/");
		await page.locator('#username').type('rahulshetty');
		await page.locator("[type='password']").type("learning");
		await page.locator("#signInBtn").click();
		const errMsg = await page.locator("[style*='block']").textContent();
		console.log(errMsg); 
		await expect(page.locator("[style*='block']")).toContainText('Incorrect username');
		
		await page.locator('#username').fill(""); //clears previous value provided in 'Username' field
		const newUserNameVal = await page.locator('#username').textContent();

		console.log("Username -->" + newUserNameVal );
		await page.locator('#username').fill("rahulshettyacademy"); //provide new value to username
		await page.locator("#signInBtn").click();
		
		console.log(await page.locator(".card-body a").nth(0).textContent()); //if from your css/xpath multiple elements are returned and you want to get the first 1 only.
		console.log(await page.locator(".card-body a").first().textContent()); //if from your css/xpath multiple elements are returned and you want to get the first 1 only.
		console.log(await page.locator(".card-body a").last().textContent()); //if from your css/xpath multiple elements are returned and you want to get the last only.
		console.log(await page.locator(".card-body a").allTextContents()); //grab all multiple elements text into a list. BUt it doesnt wait to populate, while the above ones 'first()','nth()' does wait.. CHeck documentation "Auto-Wait"
	}
);
```

### Dynamic wait till all the APIs of the website is in Idle state or all the network calls are made.
1. Wait for API calls to be done
```example.spec.js
const {test} = require(`@playwright/test`);
const {expect} =  require(`@playwright/test`);


test.only ('First Playwright TestCase', async ({browser,page})=>
	{
		await page.goto("https://rahulshettyacademy.com/client/");
		await page.locator('#userEmail').fill('anshika@gmail.com');
		await page.locator("#userPassword").type("Iamking@000");
		await page.locator("[value='Login']").click();

		await page.waitForLoadState('networkidle'); //waits till all API calls are made and are successful
		console.log(await page.locator(".card-body b").allTextContents());
	}
);
```

### Dynamic wait when there are no API calls
1. Wait for page to be loaded 'Page Navigation Wait' when no API
```example.spec.js
const {test} = require(`@playwright/test`);
const {expect} =  require(`@playwright/test`);


test.only ('First Playwright TestCase', async ({browser,page})=>
	{
		await page.goto("https://rahulshettyacademy.com/loginpagePractise/");
		await page.locator('#username').type('rahulshetty');
		await page.locator("[type='password']").type("learning");
		await page.locator("#signInBtn").click();
		const errMsg = await page.locator("[style*='block']").textContent(); 
		console.log(errMsg); 
		await expect(page.locator("[style*='block']")).toContainText('Incorrect username');

		await page.locator('#username').fill("");
		await page.locator('#username').fill("rahulshettyacademy");

		/*
			Here we are saying that waitForNavigation wait should be used after user clicks on SignINBtn.
			Both are interlinked hence it is put in a block which is a Promise
			Terms - Race condition
		*/
		await Promise.all(
			[
				page.waitForNavigation(),
				page.locator("#signInBtn").click();
			]
		)

	}
);
```
