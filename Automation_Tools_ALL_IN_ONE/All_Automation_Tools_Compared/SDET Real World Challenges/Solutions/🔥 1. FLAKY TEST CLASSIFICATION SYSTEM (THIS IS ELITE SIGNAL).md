## 🔧 Capture Failure Automatically (Playwright)

```ts
test.afterEach(async ({}, testInfo) => {
  if (testInfo.status !== testInfo.expectedStatus) {
    const error = testInfo.error?.message || '';

    let type = classifyError(error);

    console.log({
      test: testInfo.title,
      type,
      error
    });
  }
});
```

## 🧠 Classification Logic (Real Insight)

```ts
function classifyError(error: string): FailureType {
  if (error.includes('Timeout')) return FailureType.TIMING;
  if (error.includes('ECONNREFUSED')) return FailureType.NETWORK;
  if (error.includes('already exists')) return FailureType.DATA;
  return FailureType.TEST_BUG;
}
```
