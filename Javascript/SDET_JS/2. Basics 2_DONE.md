## Function

#### 1. Function Declarations
Note - I can call function before declaring it and it will still work because of a process called `hoisting`. 
```Javascript
'use strict';
let name = "Dingus";
// Function Declaration
function logger(name){
	console.log(`my name is ${name}`);
	return 10;
}

console.log(logger(name));
```

Also if a function which doesn't have `return` returns `undefined` by default
```Javascript
function demo(){
	//code goes here.
}
console.log(demo());
```

#### 2. Function Expression or Anonymous Function
```Javascript
'use strict';

// Function Expression
const calculateAge = function(birthYear) {
	return 2026 - birthYear;
}

const age = calculateAge(2021);
console.log(age);
```

#### 3. Arrow Function
```Javascript
'use strict';

//Arrow Function
const calcAge = birthYear => 2026 -  birthYear;
console.log(calcAge(2020));

//Another Example of Arrow Function
const calcRetirement  = (birthYear, firstName) => {
	const age = 2026 - birthYear;
	const retirementAge = 65 - age;
	return `${firstName} retires in ${retirementAge}`;
};

console.log(calcRetirement(2010, 'John'));

```

## Arrays
```Javascript
'use strict';

const friends = ['Tom', 'Bob', 'Dylan'];

console.log(friends[friends.length-1]);

const firstName = 'Tim';

const me = [firstName, 2026 - 1990, "Apples", friends];

```

#### Array Methods
```Javascript
'use strict';

const friends = ['Tom', 'Tim', 'Bob'];
console.log(friends);

// Add elements 
const newLength = friends.push('Mina');
console.log(newLength); // returns new length
console.log(friends);

// Remove Elements
friends.pop(); 
console.log(friends);
friends.shift();
console.log(friends);

// Find elements
console.log(friends.indexOf('Tim')); // returns index
console.log(friends.includes('Naz')); // false
```

## Objects
```Javascript
'use strict';
//Array
const Tim = [1990, "Tim Horton", "MBA Grad", "Football"];

// Object (Literal syntax)
const timData = {
	age: 1990,
	name: "Tim Horton",
	class: "MBA Grad",
	hobby: "Football"
};

console.log(timData);
console.log(typeof(timData));
console.log(timData.age, timData.hobby);
console.log(timData['name']); //Expression can be used here. More flexible.

//Example of why its more flexible
const userInput = prompt('what you want to know about Tim? (age, name, class, hobby)');
console.log(userInput);
if(timData[userInput]){
	console.log(timData[userInput]);
}else{
console.log('Wrong request!');
}

//Adding more values to object
timData.location = 'USA';
timData['onTwitter'] = false;
```

##### Object Methods
```Javascript
'use strict';

// Object (Literal syntax)
const timData = {
	age: 1990,
	name: "Tim Horton",
	class: "MBA Grad",
	hobby: "Football",
	calcAge: function (birthYear) {return 2025-birthYear;},
	calcAge2: function () {console.log(this); return 2025 - this.age;},
	calcAge3: function () {this.timAge = 2025 - this.age;} // useful if you dont want to calculate age again and again eg if its a resource heavy computation.
};

console.log(timData.calcAge(2000));
console.log(timData['calcAge'](2000));

console.log(timData.calcAge2());
console.log(timData['calcAge2']());

console.log(timData.calcAge3());
console.log(timData.timAge);
```

## Iteration - `for` loop
```Javascript
'use strict';

for(let i = 0 ; i < 5 ; i++){
	console.log(i);
}
```

#### Loop Arrays
```Javascript
'use strict';

const friends = [
	'Tim',
	'Kim',
	'Nathan',
	'John',
	'Peter',
	'Sam'
];

for (let i=0 ; i < friends.length ; i++){
	console.log(friends[i]);
}

// Loop backwards
for (let i=friends.length-1 ; i >= 0 ; i--){
	console.log(friends[i]);
}
```

## Iteration -`while`loop
```Javascript
'use strict';

const friends = [
	'Tim',
	'Kim',
	'Nathan',
	'John',
	'Peter',
	'Sam'
];

let i = 0;
while ( i < friends.length){
	console.log(friends[i]);
	i++;
}
```
